# MCP Settings UI Redesign Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the flat MCP settings tab with a two-view tab (server list → per-server permission sub-screen) that replicates Claude Code's tool-permissions UI, backed by a 3-group default/deviation permission model.

**Architecture:** Core `mcp-registry.ts` gains a 3rd tool group (`interactive`), a per-group default, and a `"default"` per-tool sentinel resolved inside the existing grant functions (so codegen is untouched). The plugin's `renderMcpTab` becomes list+detail views; the detail view renders per-group default pulldowns and per-tool 3-icon segmented controls.

**Tech Stack:** TypeScript, Vitest, Obsidian plugin API (Setting/DropdownComponent/ExtraButton/setIcon), esbuild.

**Spec:** `docs/specs/2026-06-09-mcp-settings-ui-redesign-design.md`

```typescript
/** Resolve a tool's effective permission: its own concrete value, or — when the
 *  tool is set to "default" — the default of its group. Pure. */
export function resolveToolPermission(server: McpRegistryServer, toolName: string): McpToolPermission {
  const t = server.tools[toolName];
  if (!t) return "blocked"; // absent = strictest, consistent with isPolicyStricter
  if (t.permission !== "default") return t.permission;
  return serverGroupDefaults(server)[t.group];
}
```

```typescript
/** Partition a server's tools by RESOLVED permission (sorted for determinism). Pure. */
export function resolveServerGrants(server: McpRegistryServer): ServerGrants {
  const allow: string[] = [], ask: string[] = [], blocked: string[] = [];
  for (const name of Object.keys(server.tools)) {
    const p = resolveToolPermission(server, name);
    (p === "allow" ? allow : p === "ask" ? ask : blocked).push(name);
  }
  const sorted = (a: string[]) => [...a].sort();
  return { allow: sorted(allow), ask: sorted(ask), blocked: sorted(blocked) };
}
```

```typescript
/** Set a GROUP's default permission and collapse every tool in that group to
 *  "default" (so they all follow the new default — deviations cleared). Immutable. */
export function applyGroupPermission(
  server: McpRegistryServer,
  group: McpToolGroup,
  permission: McpToolPermission,
): McpRegistryServer {
  const tools: Record<string, McpRegistryTool> = {};
  for (const [name, t] of Object.entries(server.tools)) {
    tools[name] = t.group === group ? { ...t, permission: "default" } : t;
  }
  return { ...server, groupDefaults: { ...serverGroupDefaults(server), [group]: permission }, tools };
}
```

```typescript
/** Set a single tool's permission. If the chosen value equals the tool's current
 *  group default, store "default" instead (collapse the deviation). No-op for an
 *  unknown tool. Immutable. */
export function setToolPermission(
  server: McpRegistryServer,
  toolName: string,
  permission: McpToolPermission,
): McpRegistryServer {
  const t = server.tools[toolName];
  if (!t) return server;
  const groupDefault = serverGroupDefaults(server)[t.group];
  const stored: McpStoredPermission = permission === groupDefault ? "default" : permission;
  return { ...server, tools: { ...server.tools, [toolName]: { ...t, permission: stored } } };
}

/** True when every tool in a group resolves to the group's default (no deviations).
 *  An empty group is uniform. Pure. */
export function groupIsUniform(server: McpRegistryServer, group: McpToolGroup): boolean {
  const def = serverGroupDefaults(server)[group];
  return Object.keys(server.tools)
    .filter((n) => server.tools[n].group === group)
    .every((n) => resolveToolPermission(server, n) === def);
}
```

```typescript
  private async renderMcpDetail(el: HTMLElement, serverName: string): Promise<void> {
    const plugin = this.plugin;
    const reg = plugin.settings.mcpRegistry[serverName];
    new Setting(el).addButton((b) =>
      b.setButtonText("‹ Back to servers").onClick(() => {
        this.mcpView = { mode: "list" };
        void this.renderMcpTab(el);
      }),
    );
    if (!reg || reg.probe.status !== "hot") {
      renderInfoBox(el, { variant: "info", title: serverName, body: "This server is not enabled or has not been probed." });
      return;
    }
    new Setting(el).setName(`${serverName} — tool permissions`).setHeading()
      .setDesc("Choose when the agent may use these tools.");
    if (reg.probe.status === "hot") {
      // (re-probe affordance, parity with the list)
    }
    const GROUPS: { key: McpToolGroup; label: string }[] = [
      { key: "read", label: "Read-only tools" },
      { key: "interactive", label: "Interactive tools" },
      { key: "write", label: "Write / delete tools" },
    ];
    for (const { key, label } of GROUPS) {
      this.renderPermissionGroup(el, serverName, key, label);
    }
  }

  /** Render one permission group: a default dropdown ("Custom" when mixed) + a
   *  3-icon segmented control per tool. */
  private renderPermissionGroup(el: HTMLElement, serverName: string, group: McpToolGroup, label: string): void {
    const plugin = this.plugin;
    const reg = plugin.settings.mcpRegistry[serverName];
    const toolNames = Object.keys(reg.tools).filter((n) => reg.tools[n].group === group).sort();
    if (toolNames.length === 0) return;

    const uniform = groupIsUniform(reg, group);
    const groupDefault = serverGroupDefaults(reg)[group];
    const head = new Setting(el).setName(`${label} (${toolNames.length})`).setHeading();
    head.addDropdown((d) => {
      d.addOption("allow", "Always allow");
      d.addOption("ask", "Permission required");
      d.addOption("blocked", "Blocked");
      if (!uniform) d.addOption("__custom__", "— Custom");
      d.setValue(uniform ? groupDefault : "__custom__");
      d.onChange(async (v) => {
        if (v === "__custom__") return; // selecting the synthetic option is a no-op
        plugin.settings.mcpRegistry[serverName] = applyGroupPermission(reg, group, v as McpToolPermission);
        await plugin.saveSettings();
        await this.renderMcpTab(el);
      });
    });

    const ICONS: { perm: McpToolPermission; icon: string; tip: string }[] = [
      { perm: "allow", icon: "circle-check", tip: "Always allow" },
      { perm: "ask", icon: "hand", tip: "Permission required" },
      { perm: "blocked", icon: "ban", tip: "Blocked" },
    ];
    for (const tool of toolNames) {
      const resolved = resolveToolPermission(reg, tool);
      const row = new Setting(el).setName(tool).setDesc(reg.tools[tool].description ?? "");
      if (resolved === "blocked") row.settingEl.addClass("perspecta-mcp-tool-blocked");
      for (const { perm, icon, tip } of ICONS) {
        row.addExtraButton((b) => {
          b.setIcon(icon).setTooltip(tip);
          if (resolved === perm) b.extraSettingsEl.addClass("perspecta-mcp-perm-active");
          b.onClick(async () => {
            plugin.settings.mcpRegistry[serverName] = setToolPermission(plugin.settings.mcpRegistry[serverName], tool, perm);
            await plugin.saveSettings();
            await this.renderMcpTab(el); // re-render so the group dropdown reflects uniform/Custom
          });
        });
      }
    }
  }
```
